//wishlist-controller.js
const Wishlist = require("../model/wishlist-model.js)

const addToWishlist = async (req,res)=>{
    try{

    const {productId} = req.body;
    cont {userId} = req.body;
    const {productDetail} = req.body;

    if(!userId || !productId || !productDetail){
        res.status(404).json({msg: "Missing Data"})
    }

    const exist = await Wishlist.findOne(userId, productId)
    if(exist){
        res.status(400).json({msg: "it is already in wishlist})
    }

    const item = await Wishlist.create({
        userId,
        productId,
        productDetail
    });

    res.status(200).json(item)
    }
    catch(error)[
        console.log(error)
    ]
    
}
